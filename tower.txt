<script>
    const tg = window.Telegram.WebApp;
    tg.expand();

    const urlParams = new URLSearchParams(window.location.search);
    let balance = parseInt(urlParams.get('bal')) || 0;
    
    const tower = document.getElementById('tower');
    const balDisplay = document.getElementById('bal-display');
    const startBtn = document.getElementById('start-btn');
    const cashoutBtn = document.getElementById('cashout-btn');

    let currentLevel = 0;
    let isGameOver = true;
    let currentBet = 0;
    const MAX_LEVELS = 8;

    function createTower() {
        tower.innerHTML = '';
        for (let i = 0; i < MAX_LEVELS; i++) {
            const row = document.createElement('div');
            row.className = 'row';
            row.id = `row-${i}`;
            for (let j = 0; j < 3; j++) {
                const cell = document.createElement('div');
                cell.className = 'cell';
                cell.onclick = () => tapCell(i, j, cell);
                row.appendChild(cell);
            }
            tower.appendChild(row);
        }
    }

    function updateUI() {
        balDisplay.innerText = `💰 ${balance}`;
    }

    function syncWithBot() {
        tg.HapticFeedback.notificationOccurred('success');
        tg.sendData(JSON.stringify({ 
            action: "sync_balance", 
            final_balance: balance 
        }));
    }

    function startGame() {
        const bet = parseInt(document.getElementById('bet-amount').value);
        if (bet > balance || bet <= 0) {
            tg.HapticFeedback.notificationOccurred('error');
            return alert("Недостаточно звезд!");
        }

        balance -= bet;
        currentBet = bet;
        currentLevel = 0;
        isGameOver = false;
        
        createTower();
        updateUI();
        document.getElementById(`row-0`).classList.add('active');

        startBtn.style.display = 'none';
        cashoutBtn.style.display = 'block';
        cashoutBtn.innerText = `ЗАБРАТЬ: ${currentBet} ⭐`;
        tg.HapticFeedback.impactOccurred('medium');
    }

    function tapCell(lvl, pos, cell) {
        if (isGameOver || lvl !== currentLevel) return;

        const minePos = Math.floor(Math.random() * 3);

        if (pos === minePos) {
            cell.classList.add('wrong');
            cell.innerText = '💣';
            tg.HapticFeedback.notificationOccurred('error');
            isGameOver = true;
            setTimeout(() => {
                startBtn.style.display = 'block';
                cashoutBtn.style.display = 'none';
                updateUI();
            }, 500);
        } else {
            cell.classList.add('correct');
            cell.innerText = '💎';
            tg.HapticFeedback.impactOccurred('light');
            
            currentLevel++;
            currentBet = Math.floor(currentBet * 1.45);
            cashoutBtn.innerText = `ЗАБРАТЬ: ${currentBet} ⭐`;

            if (currentLevel >= MAX_LEVELS) {
                balance += currentBet;
                updateUI();
                setTimeout(syncWithBot, 500);
            } else {
                document.querySelectorAll('.row').forEach(r => r.classList.remove('active'));
                document.getElementById(`row-${currentLevel}`).classList.add('active');
            }
        }
    }

    startBtn.onclick = startGame;
    
    cashoutBtn.onclick = () => {
        if (!isGameOver) {
            balance += currentBet;
            updateUI();
            syncWithBot(); // Авто-сохранение и выход
        }
    };

    createTower();
    updateUI();
</script>
